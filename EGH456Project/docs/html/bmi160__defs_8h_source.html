678/* BIT SLICE GET AND SET FUNCTIONS */
679#define BMI160_GET_BITS(regvar, bitname) \
680 ((regvar & bitname##_MSK) >> bitname##_POS)
681#define BMI160_SET_BITS(regvar, bitname, val) \
682 ((regvar & ~bitname##_MSK) | \
683 ((val << bitname##_POS) & bitname##_MSK))
684
685#define BMI160_SET_BITS_POS_0(reg_data, bitname, data) \
686 ((reg_data & ~(bitname##_MSK)) | \
687 (data & bitname##_MSK))
688
689#define BMI160_GET_BITS_POS_0(reg_data, bitname) (reg_data & (bitname##_MSK))
690
692#define BMI160_SET_LOW_BYTE UINT16_C(0x00FF)
693#define BMI160_SET_HIGH_BYTE UINT16_C(0xFF00)
694
695#define BMI160_GET_LSB(var) (uint8_t)(var & BMI160_SET_LOW_BYTE)
696#define BMI160_GET_MSB(var) (uint8_t)((var & BMI160_SET_HIGH_BYTE) >> 8)
